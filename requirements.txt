# Mac / Apple Silicon (recommended first try — no companion wheels needed)
# =============================================================================
#    pip install torch torch_geometric h5py pyyaml numpy tqdm scikit-learn
#
# Verify MPS is available:
#    python -c "import torch; print('mps', torch.backends.mps.is_available())"
#
# =============================================================================
# NVIDIA GPU (fast path — companion wheels optional but recommended)
# =============================================================================
# Adjust the CUDA version (cu121) to match your driver: check with `nvidia-smi`.
#
# 1) PyTorch (pick the line matching your CUDA):
#    pip install torch --index-url https://download.pytorch.org/whl/cu121
#
# 2) PyTorch Geometric (minimal — works without companion wheels):
#    pip install torch_geometric
#
# 3) Optional speed wheels (must match torch & CUDA; skip if install fails):
#    pip install pyg_lib torch_scatter torch_sparse \
#        -f https://data.pyg.org/whl/torch-2.3.0+cu121.html
#    (replace 2.3.0+cu121 with your actual `python -c "import torch;print(torch.__version__)"`)
#
# 4) Utilities:
#    pip install h5py pyyaml numpy tqdm scikit-learn

# Verify GPU is visible:
#    python -c "import torch; print(torch.cuda.is_available(), torch.cuda.get_device_name(0))"
